function startDeviceOrientation(){
  if(!window.DeviceOrientationEvent)return;
  const attach=()=>{
    window.addEventListener('deviceorientation',e=>{
      if(e.webkitCompassHeading!=null)deviceHeading=e.webkitCompassHeading;
      else if(e.absolute===true && e.alpha!=null)deviceHeading=(360-e.alpha)%360;
      else if(e.alpha!=null)deviceHeading=(360-e.alpha)%360;
    },true);
    startCompassAnimation();
  };
  if(typeof DeviceOrientationEvent.requestPermission==='function'){
    DeviceOrientationEvent.requestPermission().then(s=>{
      if(s==='granted'){attach();toast('Phone compass enabled — rotate your phone to move the arrow.');}
      else toast('Compass permission not granted. GPS heading will still work while moving.',true);
    }).catch(()=>toast('Compass permission was blocked. Tap GPS again and allow Motion & Orientation.',true));
  }else attach();
}
function desiredHeading(){
  const moving=lastGpsSpeed>2.5;
  if(moving && Number.isFinite(lastGpsHeading))return lastGpsHeading;
  if(Number.isFinite(deviceHeading))return deviceHeading;
  if(Number.isFinite(lastGpsHeading))return lastGpsHeading;
  return currentHeading||0;
}
function updateGpsArrow(){
  if(!gpsMarker)return;
  const el=gpsMarker.getElement();
  if(!el)return;
  const target=desiredHeading();
  let diff=target-currentHeading;
  while(diff>180)diff-=360;
  while(diff<-180)diff+=360;
  currentHeading=(currentHeading+diff*0.22+360)%360;
  const arrow=el.querySelector('.gpsArrow');
  if(arrow)arrow.setAttribute('transform',`rotate(${currentHeading.toFixed(1)} 50 50)`);
  const txt=el.querySelector('.gpsSpeedText');
  if(txt)txt.textContent=String(Math.round(lastGpsSpeed||0));
  if(lastFix)$('statMove').textContent=`${Math.round(lastGpsSpeed||0)} mph / ${Math.round(currentHeading)}°`;
}
function startCompassAnimation(){
  if(compassAnimationStarted)return;
  compassAnimationStarted=true;
  const loop=()=>{updateGpsArrow();requestAnimationFrame(loop)};
  requestAnimationFrame(loop);
}

async function loginSpotterNetwork(){
  const u=document.getElementById('snUser').value.trim();
  const p=document.getElementById('snPass').value;
  const st=document.getElementById('loginStatus');
  if(!u||!p){st.textContent='Enter your SpotterNetwork username and password.';return}
  st.textContent='Logging in…';
  startGPSOnly();
  try{
    const r=await fetch('https://www.spotternetwork.org/login',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({username:u,password:p})});
    const j=await r.json();
    if(!j?.success||!j?.id){st.textContent='Login failed. Check username/password.';toast('SpotterNetwork login failed',true);return}
    APP_ID=j.id;
    st.textContent='Connected. Your location will stay on the map and upload every 30 seconds.';
    closeLogin();
    toast('SpotterNetwork connected — your spotter location is ON.');
    if(sendTimer)clearInterval(sendTimer);
    sendTimer=setInterval(()=>{if(lastFix)sendSpotterLocation(lastFix)},30000);
    if(lastFix)sendSpotterLocation(lastFix);
  }catch(e){st.textContent='Login request failed: '+e.message;toast('SpotterNetwork login request failed',true)}
}

  const lat=pos.coords.latitude,lon=pos.coords.longitude,mph=(pos.coords.speed||0)*2.23694,dir=pos.coords.heading||deviceHeading||0,ts=nowUtc();

  ownSpotterMarker.setLatLng([lat,lon]);
  try{
    const r=await fetch('https://www.spotternetwork.org/positions/update',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({id:APP_ID,report_at:ts,lat,lon,elev:0,mph,dir,active:1,gps:1})});
    if(r.ok){toast('Your SpotterNetwork position sent: '+ts+' UTC');}

}
